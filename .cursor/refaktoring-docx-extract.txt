// 1. Yangi fayl yaratish: src/modules/orders/domain/order.pricing.ts
import { prisma } from '../../../config/database';
import Decimal from 'decimal.js';
export async function getProductPrice(tenantId, productId, priceType) {
  const price = await prisma.productPrice.findFirst({
    where: { tenant_id: tenantId, product_id: productId, price_type: priceType }
  });
  return price ? new Decimal(price.price) : new Decimal(0);
}
export function applyDiscount(price, discountPercent) {
  const discount = new Decimal(discountPercent).div(100);
  return price.mul(new Decimal(1).minus(discount));
}
export function roundOrderMoney(amount) {
  return new Decimal(amount).toDecimalPlaces(2, Decimal.ROUND_HALF_UP);
}

import { prisma } from '../../../config/database';
import { paginationParams } from '../../../lib/pagination';
export async function listOrdersPaged(tenantId, filters, page, limit) {
  const where = { tenant_id: tenantId, ...buildWhereClause(filters) };
  const [items, total] = await Promise.all([
    prisma.order.findMany({ where, ...paginationParams(page, limit) }),
    prisma.order.count({ where })
  ]);
  return { items, total, page, limit };
}
export async function getOrderDetail(tenantId, orderId) {
  return prisma.order.findFirst({
    where: { id: orderId, tenant_id: tenantId },
    include: { order_items: true, client: true, agent: true }
  });
}

import { prisma } from '../../../config/database';
import { appendTenantAuditEvent } from '../../../lib/tenant-audit';
const ALLOWED_TRANSITIONS = {
  pending: ['confirmed', 'cancelled'],
  confirmed: ['in_delivery', 'cancelled'],
  in_delivery: ['delivered', 'returned'],
  delivered: ['completed'],
};
export async function updateOrderStatus(tenantId, orderId, newStatus, actorUserId) {
  const order = await prisma.order.findFirst({
    where: { id: orderId, tenant_id: tenantId }
  });
  if (!order) throw new Error('NOT_FOUND');
  if (!ALLOWED_TRANSITIONS[order.status]?.includes(newStatus)) {
    throw new Error('INVALID_TRANSITION');
  }
  await prisma.order.update({
    where: { id: orderId }, data: { status: newStatus }
  });
  await appendTenantAuditEvent({ ... });
}

6.2 Pull Request Strategiyasi
Katta PR emas — kichik-kichik PR
Yomon: Bitta PR — 7 ta fayl, 2775 qator o'zgartirish → code review qiyin.
Yaxshi: Har fayl — alohida PR:
  PR #1: order.pricing.ts (200 qator) → tez review, merge.
  PR #2: order.bonus.ts (250 qator) → tez review, merge.
  ...
Har PR — 200-400 qator o'zgartirish. Review 10-15 daqiqa.
6.3 Rollback Strategiyasi
Har bosqichdan keyin backup saqlanadi: orders.service.backup.ts

Production'ga chiqishdan oldin: staging'da 1 hafta test.
Production'da muammo bo'lsa — feature flag bilan yangi koddan eski kodga qaytish.
